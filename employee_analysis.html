import matplotlib.pyplot as plt
import pandas as pd
from io import StringIO

data = """employee_id,department,region,performance_score,years_experience,satisfaction_rating
EMP001,Operations,Latin America,63.85,4,4.9
EMP002,R&D,Africa,87.87,10,4.2
EMP003,HR,Latin America,85.49,7,3
EMP004,R&D,Africa,74.15,5,3.7
EMP005,HR,Asia Pacific,61.19,7,4.9
"""
df = pd.read_csv(StringIO(data))
finance_count = (df['department'] == 'Finance').sum()
print(finance_count)

df['department'].value_counts().plot(kind='bar')
plt.title('Department Distribution')
plt.xlabel('Department')
plt.ylabel('Frequency')
plt.show()
